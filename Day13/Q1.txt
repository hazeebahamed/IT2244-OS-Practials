Q5) Fibbonacci Series

Code:

#include<stdio.h>
int main(){
	int num;
	printf("Enter number: ");
	scanf("%d", &num);
	
	int x=0;
	int y=1;
	
	for(int i=1; i<=num; i++){
		printf("%d ",x);
		int z = x+y;
		x = y;
		y = z;
	}
	printf("\n");
return 0;
}

Q9)  Write a C program that takes a binary number (as an integer) as input and converts it to its decimal equivalent.

Code:

#include <stdio.h>

int main() {
    int binary, decimal = 0, base = 1, remainder;

    printf("Enter a binary number: ");
    scanf("%d", &binary);

    while (binary > 0) {
        remainder = binary % 10;
        decimal += remainder * base;
        base *= 2;
        binary /= 10;
    }

    printf("Decimal equivalent: %d\n", decimal);
    return 0;
}

Q10) Write a C program that:
Accepts an array of integers from the user.
Finds and displays the maximum and minimum values in the array.


Code:


#include <stdio.h>

int main() {
    int arr[100], n, i, max, min;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter %d integers:\n", n);
    for (i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    max = min = arr[0];

    for (i = 1; i < n; i++) {
        if (arr[i] > max)
            max = arr[i];
        if (arr[i] < min)
            min = arr[i];
    }

    printf("Maximum: %d\n", max);
    printf("Minimum: %d\n", min);
    return 0;
}

Q11) C program to generate Pascal's Triangle.
Output:
no of rows =5
          1
        1   1
      1   2   1
    1   3   3   1
  1   4   6   4   1
	

Code:


#include <stdio.h>

int factorial(int n) {
    int f = 1;
    for (int i = 1; i <= n; i++)
        f *= i;
    return f;
}

int combination(int n, int r) {
    return factorial(n) / (factorial(r) * factorial(n - r));
}

int main() {
    int rows, i, j, space;

    printf("Enter number of rows: ");
    scanf("%d", &rows);

    for (i = 0; i < rows; i++) {
        for (space = 0; space < rows - i - 1; space++)
            printf("  ");
        for (j = 0; j <= i; j++)
            printf("%4d", combination(i, j));
        printf("\n");
    }

    return 0;
}
